import React, { useState, useEffect } from "react";

const Admin = () => {
  const [formLink, setFormLink] = useState("");
  const [inscricoes, setInscricoes] = useState([]);
  const [formsRespostas, setFormsRespostas] = useState([]);
  const [unificados, setUnificados] = useState([]);
  const [loading, setLoading] = useState(true);

  // ðŸ”¹ Carregar config do Forms
  useEffect(() => {
    fetch("http://localhost:4000/api/forms-link")
      .then((res) => res.json())
      .then((data) => data.formsLink && setFormLink(data.formsLink))
      .catch((err) => console.error("Erro ao carregar link do Forms:", err));
  }, []);

  // ðŸ”¹ Salvar link do Forms
  const handleSaveLink = async () => {
    try {
      const response = await fetch("http://localhost:4000/api/forms-link", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ formsLink: formLink }),
      });

      const result = await response.json();
      if (result.success) {
        alert("âœ… Link salvo com sucesso!");
      } else {
        alert("âš ï¸ Erro ao salvar o link.");
      }
    } catch (error) {
      console.error("Erro ao salvar link:", error);
      alert("âŒ Erro na comunicaÃ§Ã£o com o servidor.");
    }
  };

  // ðŸ”¹ Carregar inscriÃ§Ãµes (etapa 1) e respostas (etapa 2)
  useEffect(() => {
    Promise.all([
      fetch("http://localhost:4000/api/inscricoes").then((r) => r.json()),
      fetch("http://localhost:4000/api/forms-respostas").then((r) => r.json()),
    ])
      .then(([insc, forms]) => {
        setInscricoes(insc);
        setFormsRespostas(forms.respostas);

        // ðŸ”„ Merge: base = etapa 1 (SQLite), complementa com Forms
        const unificados = insc.map((i) => {
          const match = forms.respostas.find(
            (f) =>
              f["Qual Ã© o seu e-mail?"] &&
              f["Qual Ã© o seu e-mail?"].trim().toLowerCase() ===
                i.email.trim().toLowerCase()
          );

          return {
            ...i,     // ðŸ‘ˆ sempre traz Nome/Telefone/Email/Evento do banco (etapa 1)
            ...match, // ðŸ‘ˆ adiciona dados do Forms (inclusive uploads)
          };
        });

        setUnificados(unificados);
      })
      .catch((err) => console.error("Erro ao carregar dados:", err))
      .finally(() => setLoading(false));
  }, []);

  return (
    <div className="p-6">
      <h1 className="text-2xl font-bold mb-4">Painel Administrativo</h1>

      {/* ðŸ”¹ Campo de link do Forms + botÃ£o Salvar */}
      <div className="mb-6">
        <label className="block font-medium">Link do Google Forms:</label>
        <input
          type="text"
          value={formLink}
          onChange={(e) => setFormLink(e.target.value)}
          className="border p-2 w-full rounded"
        />
        <button
          onClick={handleSaveLink}
          className="mt-2 px-4 py-2 bg-blue-600 text-white rounded hover:bg-blue-700"
        >
          Salvar
        </button>
      </div>

      {/* ðŸ”¹ Tabela Unificada */}
      <h2 className="text-xl font-semibold mb-2">Tabela Unificada</h2>

      {loading ? (
        <p>Carregando...</p>
      ) : (
        <table className="border-collapse border w-full mt-2">
          <thead>
            <tr className="bg-gray-200">
              <th className="border px-2">Nome</th>
              <th className="border px-2">Telefone</th>
              <th className="border px-2">E-mail</th>
              <th className="border px-2">TÃ­tulo do evento</th>
              <th className="border px-2">Documentos</th>
            </tr>
          </thead>

          <tbody>
            {unificados.map((u, idx) => {
              // ðŸ”¹ Buscar uploads (apenas do Forms â†’ {url, name} ou links diretos)
              const uploads = Object.keys(u).filter(
                (k) =>
                  u[k] &&
                  ((typeof u[k] === "string" && u[k].startsWith("http")) ||
                    (typeof u[k] === "object" && u[k].url))
              );

              return (
                <tr key={idx}>
                  {/* ðŸ‘‡ Campos da etapa 1 (SQLite) */}
                  <td className="border px-2">{u.nome}</td>
                  <td className="border px-2">{u.telefone}</td>
                  <td className="border px-2">{u.email}</td>
                  <td className="border px-2">{u.evento_nome}</td>

                  {/* ðŸ‘‡ Documentos do Forms */}
                  <td className="border px-2">
                    {uploads.length > 0 ? (
                      uploads.map((key, i) => {
                        const valor = u[key];
                        const link =
                          typeof valor === "string" ? valor : valor.url;
                        const nome =
                          typeof valor === "string"
                            ? `ðŸ“¥ Documento ${i + 1}`
                            : valor.name;

                        return (
                          <a
                            key={i}
                            href={link}
                            target="_blank"
                            rel="noopener noreferrer"
                            className="inline-block mr-2 px-3 py-1 bg-green-600 text-white rounded hover:bg-green-700"
                          >
                            {nome}
                          </a>
                        );
                      })
                    ) : (
                      <span>-</span>
                    )}
                  </td>
                </tr>
              );
            })}
          </tbody>
        </table>
      )}
    </div>
  );
};

export default Admin;
